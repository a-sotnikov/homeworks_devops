playbook:

```
---
- name: Setup Nginx
  hosts: yandex_cloud
  become: true

  tasks:
  - name: Install Nginx
    ansible.builtin.apt:
      package: nginx
      state: present
      update_cache: true

  - name: Create nginx user
    ansible.builtin.user:
      name: www-data
      group: www-data
      create_home: false
      system: true

  - name: Create www dir
    ansible.builtin.file:
      path: /data/www
      state: directory
      owner: www-data
      mode: 0770

  - name: Setup index.html
    ansible.builtin.copy:
      src: "./index.html"
      dest: "/data/www/index.html"
      owner: www-data
      mode: 0770

  - name: Setup nginx config
    ansible.builtin.copy:
      src: "./nginx.conf"
      dest: "/etc/nginx/nginx.conf"
      mode: 0444
    notify: Restart nginx

  - name: Enable and start nginx
    ansible.builtin.systemd:
      service: nginx
      state: started
      enabled: true

  handlers:
    - name: Restart nginx
      ansible.builtin.systemd:
        service: nginx
        state: restarted
```
